> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement `luna-sdk`, `luna-proxy`, and `luna-agent` as a Go workspace with mTLS, transaction-based SSH cert signing, Telegram OOB approval, and Vault SSH CA integration per `docs/superpowers/specs/2026-05-21-luna-core-design.md`.

**Architecture:** Three modules in one workspace — publishable SDK (HTTP client + crypto), central proxy (auth pipeline + approval + Vault), agent daemon (SSH agent protocol wrapping SDK). Sign flow: `POST /api/v1/ssh/sign` → `GET .../wait`.

**Tech Stack:** Go 1.22+, `golang.org/x/crypto/ssh`, `github.com/oklog/ulid`, standard library `crypto/tls`, Linux `SO_PEERCRED` (build tag `linux`), docker-compose for E2E.

**Spec:** `docs/superpowers/specs/2026-05-21-luna-core-design.md`

```go
package sdk

import (
    "crypto/ed25519"
    "encoding/hex"
    "fmt"

    "golang.org/x/crypto/ssh"
)

func challenge(user, ip string, ts int64) []byte {
    return []byte(fmt.Sprintf("%s:%s:%d", user, ip, ts))
}

func SignPoP(pub ssh.PublicKey, priv ed25519.PrivateKey, user, ip string, ts int64) (string, error) {
    msg := challenge(user, ip, ts)
    sig := ed25519.Sign(priv, msg)
    sshSig := &ssh.Signature{Format: pub.Type(), Blob: sig}
    return hex.EncodeToString(sshSig.Blob), nil
}

func VerifyPoP(pub ssh.PublicKey, user, ip string, ts int64, sigHex string) error {
    blob, err := hex.DecodeString(sigHex)
    if err != nil {
        return err
    }
    sshSig := &ssh.Signature{Format: pub.Type(), Blob: blob}
    return pub.Verify(challenge(user, ip, ts), sshSig)
}
```

```go
package sign

import (
    "crypto/hmac"
    "crypto/sha256"
    "crypto/tls"
)

const exporterLabel = "luna-request-hmac"

func ComputeBodyHMAC(conn *tls.Conn, body []byte) ([]byte, error) {
    key, err := conn.ExportKeyingMaterial(exporterLabel, nil, 32)
    if err != nil {
        return nil, err
    }
    mac := hmac.New(sha256.New, key)
    mac.Write(body)
    return mac.Sum(nil), nil
}

func FormatMACHeader(mac []byte) string {
    return hex.EncodeToString(mac)
}
```
